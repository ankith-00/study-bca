FACTOR

The factor is a number or algebraic expression that divides another number or expression evenly i.e its remainder is 0. (or) factors are small numbers when multiplied gives other numbers
Example : 3 and 6 are factors of 12 because 12 / 3 = 4




FACTORING

Factoring or Factorization is nothing but writing a number as the product of smaller numbers,It is the decomposition of a number (or) mathematical objects into smaller or simpler numbers/objects. In simple words, factoring is the process of finding the factors.

PROGRAM : Finding GCD [ Greatest Common Divisor ]

 #include < stdio.h >

int findGCD(int a, int b)
{
  while (a != b) {
    if (a > b) {
      a -= b; 
    }
    else{
      b -= a;
    }
  }
  return a;
}

void main() 
{
  int num1, num2;
  clrscr();
    
  printf("Enter two numbers   : ");
  scanf("%d %d", &num1, &num2);
    
  printf("GCD of %d and %d is : %d",num1, num2,findGCD(num1, num2));
    
  getch();

}

OUTPUT


 Enter two numbers : 20 18
GCD of 20 and 18 is : 2

PROGRAM : To find the Power of a number using Exponentiation by Squaring Method. 

 #include < stdio.h >
#include

int power (int x, int n)
{
  int product = 1;
  clrscr();

  if(n == 0)    // if power is zero then     
    return 1;

  while (n > 0)
  {
    if (n % 2 == 1)
      product = product * x;
      n = n/2;
      x = x * x;
      return product;
  }
}

PROGRAM : C program to reverse a given Array

  for(i=0; i<n; i++)
  {
    scanf ("%d", &a[i]);
    i=0;
    j=n-1;

    while (i<j)
    {
      temp = a[i];
      a[i] = a[j];

PROGRAM : C program to Remove the Duplicates from an Array. 

 #include < stdio.h >

int removeDuplicates(int arr[], int n) 
{
  if (n == 0 || n == 1)
  {
      return n;
  }

  int temp[n];
  int j = 0;

  for (int i = 0; i < n - 1; i++) 
  {
      if (arr[i] != arr[i + 1])
      {
          temp[j++] = arr[i];
      }
  }

  temp[j++] = arr[n - 1];

  for (int i = 0; i < j; i++) 
  {
      arr[i] = temp[i];
  }

  return j;
}


void main() 
{
  int arr[] = {1, 2, 2, 3, 4, 4, 4, 5};
  clrscr();

  printf("\nArray Before removing duplicates : ");

  for (int i = 0; i < sizeof(arr) / sizeof(arr[0]); i++)
  {
    printf("%3d ", arr[i]);
  }

  int n = sizeof(arr) / sizeof(arr[0]);
  n = removeDuplicates(arr, n);
      
  printf("\nArray After removing duplicates : ");

  for (int i = 0; i < n; i++) 
  {
    printf("%3d ", arr[i]);
  }
    
  getch();

}

OUTPUT


Array Before removing duplicates : 1 2 2 3 4 4 4 5  
Array After removing duplicates : 1 2 3 4 5



PROGRAM : C Program to partition an array 

 #include < stdio.h >

#include
void partition(int arr[], int low, int high) 
{
  int pivot = arr[high];
  int i = low - 1;

  for (int j = low; j < high; j++)     
  {
      if (arr[j] <  pivot) 
      {
          i++;
          int temp = arr[i];
          arr[i] = arr[j];
          arr[j] = temp;
      }
  }

  int temp = arr[i + 1];
  arr[i + 1] = arr[high];
  arr[high] = temp;
}

void main() 
{
  int arr[] = {4, 7, 2, 5, 1, 8, 3};
  clrscr();
  int n = sizeof(arr) / 
  sizeof(arr[0]);
  partition(arr, 0, n - 1);

  printf("Partitioned array : ");
  for (int i = 0; i < n; i++) 
  {
    printf("%d ", arr[i]);
  }
  getch();

}

OUTPUT


Partitioned array : 2 1 3 5 7 8 4
